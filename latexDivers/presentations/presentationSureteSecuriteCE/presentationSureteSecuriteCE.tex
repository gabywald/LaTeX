\subsection{Ing{\'e}nierie Sociale (1)}
\begin{frame}
	\frametitle{Ing{\'e}nierie Sociale (1)}
	\begin{itemize}

		\item V{\'e}rification (syst{\'e}matique) des informations, interlocuteurs connus et reconnus ;
